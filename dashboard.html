document.addEventListener("DOMContentLoaded", function() { const role = sessionStorage.getItem('role'); if (role) { // แสดงข้อความตามบทบาท const roleMessage = document.getElementById('role-message'); roleMessage.textContent = `คุณคือ: ${role}`; // ข้อมูลจำลอง const pendingOrders = 10; // จำนวนคำสั่งซื้อที่รอดำเนินการ const approvedOrders = 5; // จำนวนคำสั่งซื้อที่ได้รับการอนุมัติแล้ว const updatedOrders = 3; // จำนวนคำสั่งซื้อที่ถูกอัปเดตแล้ว // อัปเดตข้อมูลในหน้า Dashboard document.getElementById('pending-orders').textContent = pendingOrders; document.getElementById('approved-orders').textContent = approvedOrders; document.getElementById('updated-orders').textContent = updatedOrders; } else { alert("คุณต้องเข้าสู่ระบบก่อน"); window.location.href = "login.html"; // หากไม่มีข้อมูลบทบาทผู้ใช้, กลับไปที่หน้า login } });
